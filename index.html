Algorithm Overview:

Problem: Find the minimum number of coins needed to make a target amount.

DP Formula: dp[i] = min(dp[i], dp[i - coin] + 1) for each coin value.

Base Case: dp[0] = 0 (zero coins needed for amount 0)

Algorithm Code

int minCoins(int coins[], int n, int amount)
{
    int dp[amount + 1];

    // Step 1: Initialize dp array
    dp[0] = 0;
    for (int i = 1; i <= amount; i++)
        dp[i] = INT_MAX;

    // Step 2: Fill dp array
    for (int i = 1; i <= amount; i++)
    {
        for (int j = 0; j < n; j++)
        {
            if (coins[j] <= i && dp[i - coins[j]] != INT_MAX)
            {
                dp[i] = (dp[i] < dp[i - coins[j]] + 1)
                        ? dp[i]
                        : dp[i - coins[j]] + 1;
            }
        }
    }

    // Step 3: Return result
    if (dp[amount] == INT_MAX)
        return -1;
    else
        return dp[amount];
}
